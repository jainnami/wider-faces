# Function to extract highest AI confidence
get_max_confidences <- function(folder_path, sample_size = 80) {
  image_files <- list.files(folder_path, full.names = TRUE, pattern = "\\.jpg$|\\.jpeg$")
  image_files <- image_files[!is.na(image_files)]
  if(length(image_files) == 0) return(c())
  
  actual_sample <- min(sample_size, length(image_files))
  sampled_files <- sample(image_files, actual_sample)
  max_confs <- numeric(actual_sample)
  
  for (i in 1:actual_sample) {
    img <- tryCatch(image_read(sampled_files[i]), error = function(e) NULL)
    if (!is.null(img)) {
      faces <- image_detect_faces(img)
      if (length(faces$detections$confidence) > 0) {
        max_confs[i] <- max(faces$detections$confidence, na.rm = TRUE)
      } else {
        max_confs[i] <- 0 
      }
    }
  }
  return(max_confs)
}

# Function to calculate Detection Rate and 95% Confidence Interval
calc_stats <- function(conf_vector, threshold) {
  n <- length(conf_vector)
  successes <- sum(conf_vector >= threshold)
  p <- successes / n
  margin_error <- 1.96 * sqrt((p * (1 - p)) / n)
  lower_bound <- max(0, (p - margin_error) * 100)
  upper_bound <- min(100, (p + margin_error) * 100)
  return(list(rate = p * 100, lower = lower_bound, upper = upper_bound))
}
